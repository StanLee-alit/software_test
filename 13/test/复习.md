    if x<=0 or y<=0:
        return "输入不符合要求"
    absValue=abs(x-y)
    if x=y:
        return "可以构建圆形或正方形"
    elif 2<absValue<=5:
        return "可以构建椭圆"
    elif absValue>5:
        return "可以构建矩形"
    elif 0<absValue<=2:
        return "可以构建长方形"
